## 一、代价函数完整设计（逐项修正）

### 第一级：硬约束剪枝层（不参与加权，直接过滤）

硬约束的处理方式是：**在A*扩展节点时，先检查硬约束，不满足则直接跳过该节点**，不进入代价计算环节。这能大幅减少无效计算。

```python
def check_hard_constraints(node, parent_node, terrain, signal_field, 
                           energy_used, energy_total, obstacles_t, config):
    """
    硬约束检查：任何一项不满足 → 返回False，该节点被剪枝
    返回: (is_valid, reject_reason)
    """
    x, y, z = node
    
    # ========== 硬约束1：地形碰撞（含膨胀安全区）==========
    if terrain[x, y, z] == 1:
        return False, "terrain_collision"
    
    # ========== 硬约束2：通信完全中断 ==========
    S_crit = config['S_crit']  # 通信中断阈值，如0.05
    if signal_field[x, y, z] < S_crit:
        return False, "comm_blackout"
    
    # ========== 硬约束3：能源耗尽 ==========
    # 剩余能量不足以返回最近充电点/起点
    energy_to_home = estimate_return_energy(node, config['home_point'])
    if energy_used + energy_to_home > config['energy_budget'] * 0.95:
        return False, "energy_critical"
    
    # ========== 硬约束4：动态障碍碰撞核心区 ==========
    R_hard = config['R_hard']  # 碰撞硬半径，如1格
    for obs in obstacles_t:
        if euclidean_dist(node, obs) < R_hard:
            return False, "obstacle_collision"
    
    # ========== 硬约束5：最小转弯半径 ==========
    if parent_node is not None:
        turn_angle = compute_turn_angle(parent_node, node)
        if turn_angle > config['max_turn_angle']:  # 如120°
            return False, "turn_radius_violation"
    
    return True, "pass"
```

### 第二级：安全软约束层（归一化[0,1]，高权重）

通过硬约束检查后，计算安全相关的连续代价。



```Python
import numpy as np

def compute_safety_cost(node, t, obstacles_t, signal_field, config):
    """
    安全软约束代价（归一化到[0,1]）
    包含：动态障碍距离衰减 + 信号质量连续惩罚
    返回: (C_obstacle_soft, C_signal_soft)
    """
    x, y, z = node
    
    # ========== 动态障碍：距离衰减软惩罚 ==========
    R_hard = config['R_hard']    # 碰撞硬半径，已在第一级过滤
    R_warn = config['R_warn']    # 预警半径，如4格
    lam = config['lambda_obs']   # 衰减系数，如0.8
    
    C_obstacle_soft = 0.0
    for obs in obstacles_t:
        dist = euclidean_dist(node, obs)
        if R_hard <= dist < R_warn:
            # 指数衰减：距离障碍越近，代价越高
            raw = np.exp(-lam * (dist - R_hard))
            C_obstacle_soft = max(C_obstacle_soft, raw)  # 取最近障碍的最大代价
        # dist >= R_warn → 代价为0，已初始化
    
    # 归一化：exp衰减的最大值为1（dist=R_hard时），已天然在[0,1]
    
    # ========== 信号质量：连续二次惩罚 ==========
    S = signal_field[x, y, z]
    S_crit = config['S_crit']  # 通信中断阈值（已在硬约束过滤）
    S_th = config['S_th']      # 通信警戒阈值，如0.3
    
    if S >= S_th:
        C_signal_soft = 0.0
    else:
        # 二次惩罚：越接近中断阈值，惩罚越陡
        C_signal_soft = ((S_th - S) / (S_th - S_crit)) ** 2
        C_signal_soft = min(C_signal_soft, 1.0)  # 上限截断
    
    return C_obstacle_soft, C_signal_soft
```

**关键改进说明**：

- **动态障碍**：原方案 Cobstacle=∞Cobstacle​=∞（整个安全半径内全部无穷大）→ 修改为内圈硬剪枝 + 外圈指数衰减，路径可以找到平滑绕行
- **信号强度**：原方案 {0,100}{0,100} 两档 → 修改为连续二次函数，消除代价阶梯跳变
- **双重惩罚修复**：多个障碍取 `max` 而非 `sum`，避免密集障碍区代价爆炸

### 第三级：效率优化层（归一化[0,1]，低权重）


```Python
def compute_efficiency_cost(node, parent_node, t, current_field, config):
    """
    效率优化代价（归一化到[0,1]）
    包含：洋流能耗 + 转向平滑 + 路径步数 + 定位误差
    返回: (C_energy, C_motion, C_length, C_localize)
    """
    x, y, z = node
    
    # ========== 洋流能耗代价（方向+强度融合）==========
    cur_u, cur_v, cur_w = current_field  # 三个3D矩阵
    v_current = np.array([cur_u[x,y,z], cur_v[x,y,z], cur_w[x,y,z]])
    
    if parent_node is not None:
        # 移动方向单位矢量
        direction = np.array(node) - np.array(parent_node)
        norm = np.linalg.norm(direction)
        if norm > 0:
            d_hat = direction / norm
        else:
            d_hat = np.array([0, 0, 0])
        
        # 洋流方向单位矢量
        v_norm = np.linalg.norm(v_current)
        if v_norm > 0:
            v_hat = v_current / v_norm
        else:
            v_hat = np.array([0, 0, 0])
        
        # 方向代价：逆流高、顺流低
        direction_cost = (1 - np.dot(v_hat, d_hat)) / 2.0  # [0, 1]
        
        # 强度代价：洋流越强，对路径干扰越大
        v_max = config['v_current_max']  # 最大洋流速度，如2.0 m/s
        strength_cost = min(v_norm / v_max, 1.0)
        
        # 融合（方向为主，强度为辅）
        beta = config.get('beta_strength', 0.3)
        C_energy = (1 - beta) * direction_cost + beta * strength_cost
    else:
        C_energy = 0.0
    
    # ========== 转向平滑代价（连续化）==========
    if parent_node is not None:
        delta_theta = compute_turn_angle_full(parent_node, node, config)
        C_motion = (delta_theta / np.pi) ** 2  # [0, 1]，急转弯惩罚更重
    else:
        C_motion = 0.0
    
    # ========== 路径步数代价 ==========
    step_count = config.get('current_step', 0)
    D_max = config['D_max']  # 最大允许步数
    C_length = min(step_count / D_max, 1.0)
    
    # ========== 定位误差代价 ==========
    # 基于深度和距离的INS累积误差模型
    depth_factor = z / config['grid_size']  # 深度越深，定位越差
    dist_from_beacon = euclidean_dist(node, config['acoustic_beacon'])
    beacon_range = config.get('beacon_range', 15)
    range_factor = min(dist_from_beacon / beacon_range, 1.0)
    
    C_localize = 0.4 * depth_factor + 0.6 * range_factor  # [0, 1]
    
    return C_energy, C_motion, C_length, C_localize


def compute_turn_angle_full(parent, current, config):
    """完整的转向角计算（需要grandparent）"""
    grandparent = config.get('grandparent', None)
    if grandparent is None:
        return 0.0
    
    v1 = np.array(current) - np.array(parent)
    v0 = np.array(parent) - np.array(grandparent)
    
    cos_theta = np.dot(v0, v1) / (np.linalg.norm(v0) * np.linalg.norm(v1) + 1e-8)
    cos_theta = np.clip(cos_theta, -1, 1)
    return np.arccos(cos_theta)
```

```Python
def compute_total_g_cost(node, parent_node, t, env_data, config, weights):
    """
    总代价函数 g(n)：三级分层计算
    
    env_data: {
        'terrain': 3D array,
        'current_field': (u, v, w),
        'obstacles_t': [(x,y,z), ...],
        'signal_field': 3D array
    }
    
    weights: {
        'w_obstacle': 0.35,  # 安全层
        'w_signal': 0.25,    # 安全层
        'w_energy': 0.20,    # 效率层
        'w_motion': 0.08,    # 效率层
        'w_length': 0.07,    # 效率层
        'w_localize': 0.05   # 效率层
    }
    """
    
    # ===================== 第一级：硬约束剪枝 =====================
    is_valid, reason = check_hard_constraints(
        node, parent_node,
        env_data['terrain'],
        env_data['signal_field'],
        config.get('energy_used', 0),
        config['energy_budget'],
        env_data['obstacles_t'],
        config
    )
    if not is_valid:
        return float('inf'), reason  # 直接返回无穷大，A*跳过该节点
    
    # ===================== 第二级：安全软约束 =====================
    C_obs, C_sig = compute_safety_cost(
        node, t,
        env_data['obstacles_t'],
        env_data['signal_field'],
        config
    )
    
    # 安全因素取加权和（注意：同类取max已在函数内处理）
    safety_cost = weights['w_obstacle'] * C_obs + weights['w_signal'] * C_sig
    
    # ===================== 第三级：效率优化 =====================
    C_eng, C_mot, C_len, C_loc = compute_efficiency_cost(
        node, parent_node, t,
        env_data['current_field'],
        config
    )
    
    efficiency_cost = (weights['w_energy'] * C_eng +
                       weights['w_motion'] * C_mot +
                       weights['w_length'] * C_len +
                       weights['w_localize'] * C_loc)
    
    # ===================== 总代价 =====================
    # 安全层代价乘以放大系数，确保优先级高于效率层
    SAFETY_PRIORITY = 2.0  # 安全层整体权重放大2倍
    g_step = SAFETY_PRIORITY * safety_cost + efficiency_cost
    
    return g_step, "pass"
```

## 三、自适应权重系统

### 场景感知权重切换器


```Python
class AdaptiveWeightManager:
    """
    场景自适应权重管理器
    根据当前环境状态动态切换权重组合
    """
    
    # 预定义场景权重模板
    WEIGHT_TEMPLATES = {
        'open_water': {
            'w_obstacle': 0.20, 'w_signal': 0.30,
            'w_energy': 0.25, 'w_motion': 0.10,
            'w_length': 0.10, 'w_localize': 0.05
        },
        'dense_reef': {
            'w_obstacle': 0.45, 'w_signal': 0.20,
            'w_energy': 0.15, 'w_motion': 0.10,
            'w_length': 0.05, 'w_localize': 0.05
        },
        'strong_current': {
            'w_obstacle': 0.20, 'w_signal': 0.15,
            'w_energy': 0.45, 'w_motion': 0.08,
            'w_length': 0.07, 'w_localize': 0.05
        },
        'comm_edge': {
            'w_obstacle': 0.15, 'w_signal': 0.50,
            'w_energy': 0.15, 'w_motion': 0.05,
            'w_length': 0.05, 'w_localize': 0.10
        },
        'low_battery': {
            'w_obstacle': 0.20, 'w_signal': 0.10,
            'w_energy': 0.50, 'w_motion': 0.05,
            'w_length': 0.10, 'w_localize': 0.05
        }
    }
    
    def __init__(self, config):
        self.config = config
        # 默认权重
        self.current_weights = self.WEIGHT_TEMPLATES['open_water'].copy()
    
    def detect_scenario(self, node, env_data, config):
        """
        根据当前节点周围的环境特征，自动判别场景类型
        返回: scenario_name, confidence
        """
        x, y, z = node
        gs = config['grid_size']
        
        # 特征1：感知范围内障碍密度
        r = config.get('perception_range', 5)
        obstacle_count = 0
        total_cells = 0
        for dx in range(-r, r+1):
            for dy in range(-r, r+1):
                for dz in range(-r, r+1):
                    nx, ny, nz = x+dx, y+dy, z+dz
                    if 0 <= nx < gs and 0 <= ny < gs and 0 <= nz < gs:
                        total_cells += 1
                        if env_data['terrain'][nx, ny, nz] == 1:
                            obstacle_count += 1
        obstacle_density = obstacle_count / max(total_cells, 1)
        
        # 特征2：局部洋流强度均值
        cur_u, cur_v, cur_w = env_data['current_field']
        local_speed = np.sqrt(
            cur_u[x,y,z]**2 + cur_v[x,y,z]**2 + cur_w[x,y,z]**2
        )
        
        # 特征3：信号强度
        signal_strength = env_data['signal_field'][x, y, z]
        
        # 特征4：剩余能量比例
        energy_ratio = 1.0 - config.get('energy_used', 0) / config['energy_budget']
        
        # ===== 场景判别（优先级从高到低）=====
        if energy_ratio < 0.25:
            return 'low_battery', 0.9
        
        if signal_strength < config['S_th'] * 1.5:
            return 'comm_edge', 0.8
        
        if obstacle_density > 0.25:
            return 'dense_reef', 0.85
        
        if local_speed > config['v_current_max'] * 0.6:
            return 'strong_current', 0.8
        
        return 'open_water', 0.7
    
    def get_weights(self, node, env_data, config):
        """
        获取当前节点的自适应权重
        支持场景间平滑过渡（避免权重突变）
        """
        scenario, confidence = self.detect_scenario(node, env_data, config)
        target_weights = self.WEIGHT_TEMPLATES[scenario]
        
        # 指数移动平均平滑过渡（避免权重在边界处剧烈跳变）
        smooth_factor = 0.3  # 越小越平滑
        for key in self.current_weights:
            self.current_weights[key] = (
                (1 - smooth_factor) * self.current_weights[key] +
                smooth_factor * target_weights[key]
            )
        
        return self.current_weights.copy(), scenario
```

## 四、改进版A*搜索引擎


```Python
import heapq
import numpy as np
from collections import defaultdict

class MultiConstraintAStar:
    """
    多约束自适应A*搜索引擎
    核心改进：
    1. 三级分层代价（硬约束剪枝 → 安全软约束 → 效率优化）
    2. 自适应权重（场景感知切换）
    3. 感知范围分层计算（近处精确、远处Transformer预测）
    4. 滚动时域重规划（每replan_interval步触发）
    """
    
    # 3D搜索方向：26邻域（上下左右前后 + 对角线）
    DIRECTIONS_26 = []
    for dx in [-1, 0, 1]:
        for dy in [-1, 0, 1]:
            for dz in [-1, 0, 1]:
                if dx == 0 and dy == 0 and dz == 0:
                    continue
                DIRECTIONS_26.append((dx, dy, dz))
    
    def __init__(self, config, transformer_model=None):
        self.config = config
        self.grid_size = config['grid_size']
        self.transformer = transformer_model  # Transformer预测器
        self.weight_manager = AdaptiveWeightManager(config)
        self.replan_interval = config.get('replan_interval', 3)  # 重规划间隔
        self.alpha = config.get('alpha', 1.2)  # 启发函数权重
        
    def heuristic(self, node, goal, env_data, t):
        """
        启发函数h(n)
        · 有Transformer → 使用学习型预测（多目标代价）
        · 无Transformer → 退化为加权欧氏距离（保底方案）
        """
        if self.transformer is not None:
            # Transformer预测：节点到终点的多目标最优代价
            h_pred = self.transformer.predict_cost_to_goal(node, goal, env_data, t)
            
            # 可容许性修正：乘以缩放因子确保不严重高估
            admissibility_factor = 0.85  # 略微低估，保证近似最优
            return h_pred * admissibility_factor
        else:
            # 退化方案：加权欧氏距离
            dist = np.sqrt(sum((a-b)**2 for a, b in zip(node, goal)))
            return dist
    
    def search(self, start, goal, env_data_sequence, start_time=0):
        """
        主搜索函数（含滚动时域重规划）
        
        参数:
            start: 起点坐标 (x, y, z)
            goal: 终点坐标 (x, y, z)
            env_data_sequence: 时序环境数据列表 [{t0数据}, {t1数据}, ...]
            start_time: 起始时间步
        
        返回:
            path: [(x,y,z), ...] 完整路径
            stats: 搜索统计信息
        """
        full_path = [start]
        current_start = start
        current_time = start_time
        total_energy = 0
        stats = {
            'total_nodes_expanded': 0,
            'replan_count': 0,
            'scenario_log': []
        }
        
        while current_start != goal:
            # 获取当前时刻的环境数据
            t = min(current_time, len(env_data_sequence) - 1)
            env_data = env_data_sequence[t]
            
            # 执行一次A*搜索（局部或全局）
            segment_path, segment_stats = self._single_search(
                current_start, goal, env_data, current_time, total_energy
            )
            
            if segment_path is None:
                print(f"⚠️ 时刻{current_time}搜索失败，尝试扩大搜索范围")
                break
            
            # 取前replan_interval步作为已确认路径
            confirmed_steps = min(self.replan_interval, len(segment_path) - 1)
            confirmed_path = segment_path[:confirmed_steps + 1]
            
            full_path.extend(confirmed_path[1:])  # 去掉重复起点
            current_start = confirmed_path[-1]
            current_time += confirmed_steps
            
            # 累计统计
            stats['total_nodes_expanded'] += segment_stats['nodes_expanded']
            stats['replan_count'] += 1
            stats['scenario_log'].append(segment_stats.get('scenario', 'unknown'))
            
            # 检查是否到达终点
            if current_start == goal:
                break
            
            # 安全检查：防止无限循环
            if stats['replan_count'] > self.config.get('max_replan', 200):
                print("⚠️ 达到最大重规划次数，终止搜索")
                break
        
        stats['path_length'] = len(full_path)
        return full_path, stats
    
    def _single_search(self, start, goal, env_data, current_time, energy_used):
        """
        单次A*搜索（核心算法）
        """
        gs = self.grid_size
        
        # Open列表：(f值, 节点坐标, 父节点坐标)
        open_list = []
        h_start = self.heuristic(start, goal, env_data, current_time)
        heapq.heappush(open_list, (h_start, start, None))
        
        # 记录表
        g_score = defaultdict(lambda: float('inf'))
        g_score[start] = 0
        came_from = {}
        closed_set = set()
        grandparent_map = {}  # 用于计算转向角
        
        nodes_expanded = 0
        current_scenario = 'unknown'
        
        while open_list:
            f_current, current, parent = heapq.heappop(open_list)
            
            # 终点检查
            if current == goal:
                path = self._reconstruct_path(came_from, current)
                return path, {
                    'nodes_expanded': nodes_expanded,
                    'scenario': current_scenario
                }
            
            if current in closed_set:
                continue
            closed_set.add(current)
            nodes_expanded += 1
            
            # 获取自适应权重
            weights, scenario = self.weight_manager.get_weights(
                current, env_data, self.config
            )
            current_scenario = scenario
            
            # 扩展26邻域
            for dx, dy, dz in self.DIRECTIONS_26:
                nx, ny, nz = current[0]+dx, current[1]+dy, current[2]+dz
                neighbor = (nx, ny, nz)
                
                # 边界检查
                if not (0 <= nx < gs and 0 <= ny < gs and 0 <= nz < gs):
                    continue
                
                if neighbor in closed_set:
                    continue
                
                # ===== 第一级：硬约束剪枝 =====
                search_config = self.config.copy()
                search_config['energy_used'] = energy_used + g_score[current]
                search_config['grandparent'] = grandparent_map.get(current, None)
                
                is_valid, reason = check_hard_constraints(
                    neighbor, current,
                    env_data['terrain'],
                    env_data['signal_field'],
                    search_config.get('energy_used', 0),
                    search_config['energy_budget'],
                    env_data['obstacles_t'],
                    search_config
                )
                if not is_valid:
                    continue  # 直接跳过，不计算代价
                
                # ===== 第二级+第三级：代价计算 =====
                step_cost, _ = compute_total_g_cost(
                    neighbor, current, current_time,
                    env_data, search_config, weights
                )
                
                # 对角线移动距离补偿
                move_dist = np.sqrt(dx**2 + dy**2 + dz**2)
                step_cost *= move_dist  # 对角线步长更长，代价按比例放大
                
                tentative_g = g_score[current] + step_cost
                
                if tentative_g < g_score[neighbor]:
                    g_score[neighbor] = tentative_g
                    came_from[neighbor] = current
                    grandparent_map[neighbor] = current
                    
                    h = self.heuristic(neighbor, goal, env_data, current_time)
                    f = tentative_g + self.alpha * h
                    heapq.heappush(open_list, (f, neighbor, current))
        
        # Open列表为空，搜索失败
        return None, {'nodes_expanded': nodes_expanded, 'scenario': current_scenario}
    
    def _reconstruct_path(self, came_from, current):
        """回溯路径"""
        path = [current]
        while current in came_from:
            current = came_from[current]
            path.append(current)
        path.reverse()
        return path
```

```Python
def generate_static_terrain(config):
    """
    静态基础层：海底地形（改进版）
    改进点：
    1. 用scipy的binary_dilation替代手动三重循环（性能提升100倍+）
    2. 确保起点/终点区域清空
    3. 控制膨胀后总障碍率不超过40%（保证路径可达）
    """

    rng = np.random.RandomState(42)

        [0, 1], size=(gs, gs, gs),

    )
    
    # 球形结构元素进行膨胀（比立方体更自然）
    struct_radius = config.SAFE_DILATION
    struct = np.zeros((2*struct_radius+1,)*3, dtype=bool)
    center = struct_radius
    for i in range(2*struct_radius+1):
        for j in range(2*struct_radius+1):
            for k in range(2*struct_radius+1):
                if (i-center)**2 + (j-center)**2 + (k-center)**2 <= struct_radius**2:
                    struct[i,j,k] = True
    
    terrain = binary_dilation(raw_terrain, structure=struct).astype(np.int8)
    
    # 清空起点和终点周围3格区域（保证可达）
    for point in [config.HOME_POINT, (gs-2, gs-2, gs-2)]:
        px, py, pz = point
        terrain[max(0,px-3):px+4, max(0,py-3):py+4, max(0,pz-3):pz+4] = 0
    
    # 检查障碍率
    obstacle_ratio = terrain.sum() / terrain.size
    print(f"  地形障碍率：{obstacle_ratio:.1%}（建议<40%）")
    
    return terrain


def generate_dynamic_current(config, t):
    """
    动态环境层：3D洋流矢量场（改进版）
    改进点：
    1. 向量化计算替代三重循环（性能提升1000倍+）
    2. 洋流速度限制在v_max范围内
    3. 加入深度衰减（深层洋流更弱，符合物理规律）
    """
    gs = config.GRID_SIZE
    v_max = config.V_CURRENT_MAX
    
    # 创建坐标网格
    x = np.arange(gs)
    y = np.arange(gs)
    z = np.arange(gs)
    X, Y, Z = np.meshgrid(x, y, z, indexing='ij')
    
    # 深度衰减因子（z越大=越深，洋流越弱）
    depth_decay = np.exp(-0.05 * Z)
    
    # 时变洋流场
    u = 1.5 * np.sin(0.2 * X + 0.1 * t) * depth_decay
    v = 1.5 * np.cos(0.2 * Y + 0.1 * t) * depth_decay
    w = 0.3 * np.sin(0.1 * Z + 0.05 * t) * depth_decay
    
    # 限制最大速度
    speed = np.sqrt(u**2 + v**2 + w**2)
    exceed_mask = speed > v_max
    if exceed_mask.any():
        scale = v_max / (speed + 1e-8)
        u[exceed_mask] *= scale[exceed_mask]
        v[exceed_mask] *= scale[exceed_mask]
        w[exceed_mask] *= scale[exceed_mask]
    
    return u, v, w


def generate_dynamic_obstacles(config, t, num_obstacles=6):
    """
    动态障碍物生成（改进版）
    改进点：时间连续性——障碍物轨迹在时间步之间平滑移动
    """
    gs = config.GRID_SIZE
    margin = config.R_WARN + 1
    
    rng = np.random.RandomState(100)  # 固定种子保证轨迹可复现
    
    # 基础位置（t=0时刻）
    base_positions = []
    for _ in range(num_obstacles):
        base_positions.append([
            rng.randint(margin, gs - margin),
            rng.randint(margin, gs - margin),
            rng.randint(margin, gs - margin)
        ])
    
    # t时刻位置 = 基础位置 + 时间偏移（正弦运动，保证连续性）
    obstacles = []
    for i, base in enumerate(base_positions):
        ox = int(base[0] + 2 * np.sin(0.3 * t + i)) % gs
        oy = int(base[1] + 2 * np.cos(0.2 * t + i * 0.5)) % gs
        oz = int(np.clip(base[2] + np.sin(0.1 * t + i), 0, gs - 1))
        obstacles.append((ox, oy, oz))
    
    return obstacles


def generate_signal_field(config):
    """
    约束掩码层：水声信号强度场（改进版）
    改进点：
    1. 向量化计算
    2. 多信标支持（未来扩展）

    z = np.arange(gs)

    signal = np.exp(-0.08 * dist) * np.maximum(1 - 0.03 * Z, 0.0)
    signal = np.clip(signal, 0.0, 1.0)
    
    return signal
```

## 六、Transformer前瞻预测网络设计


```Python
import torch
import torch.nn as nn

class OceanPathTransformer(nn.Module):
    """
    水下路径规划Transformer
    
    功能：
    1. 输入时空地图序列 → 输出学习型启发函数h(n)
    2. 同时输出自适应权重向量W(t)（多任务学习）
    
    架构：时空编码器 + 双头解码器
    """
    
    def __init__(self, config):
        super().__init__()
        
        self.grid_size = config.get('grid_size', 20)
        self.d_model = config.get('d_model', 128)
        self.nhead = config.get('nhead', 8)
        self.num_layers = config.get('num_layers', 4)
        self.predict_steps = config.get('predict_steps', 5)
        
        # ===== 输入编码 =====
        # 每个节点特征维度：地形(1) + 洋流(3) + 信号(1) + 障碍距离(1) = 6
        self.input_dim = 6
        
        # 空间嵌入：将3D坐标映射到高维空间
        self.spatial_embed = nn.Linear(3, self.d_model // 2)
        
        # 特征嵌入：将环境特征映射到高维空间  
        self.feature_embed = nn.Linear(self.input_dim, self.d_model // 2)
        
        # 时间编码（可学习）
        self.time_embed = nn.Embedding(20, self.d_model)  # 最多20个时间步
        
        # ===== Transformer编码器 =====
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=self.d_model,
            nhead=self.nhead,
            dim_feedforward=self.d_model * 4,
            dropout=0.1,
            batch_first=True
        )
        self.transformer_encoder = nn.TransformerEncoder(
            encoder_layer,
            num_layers=self.num_layers
        )
        
        # ===== 解码头1：启发函数h(n)预测 =====
        self.h_head = nn.Sequential(
            nn.Linear(self.d_model, self.d_model // 2),
            nn.ReLU(),
            nn.Linear(self.d_model // 2, 1),
            nn.Softplus()  # 保证h(n) >= 0
        )
        
        # ===== 解码头2：自适应权重预测 =====
        self.weight_head = nn.Sequential(
            nn.Linear(self.d_model, self.d_model // 2),
            nn.ReLU(),
            nn.Linear(self.d_model // 2, 6),  # 6个权重
            nn.Softmax(dim=-1)  # 归一化为概率分布（权重和=1）
        )
        
        # ===== 解码头3：未来洋流场预测（辅助任务）=====
        self.current_pred_head = nn.Sequential(
            nn.Linear(self.d_model, self.d_model // 2),
            nn.ReLU(),
            nn.Linear(self.d_model // 2, 3)  # 预测(u,v,w)
        )
    
    def encode_node(self, coords, features, time_step):
        """
        将节点信息编码为Transformer输入token
        
        coords: (batch, seq_len, 3) —— 路径上各节点坐标
        features: (batch, seq_len, 6) —— 各节点环境特征
        time_step: (batch, seq_len) —— 时间步
        """
        spatial = self.spatial_embed(coords.float())    # (B, L, d/2)
        feat = self.feature_embed(features.float())     # (B, L, d/2)
        time = self.time_embed(time_step)               # (B, L, d)
        
        # 拼接空间+特征，再加上时间编码
        token = torch.cat([spatial, feat], dim=-1) + time  # (B, L, d)
        return token
    
    def forward(self, coords, features, time_steps, goal_coords):
        """
        前向传播
        
        返回:
            h_pred: (batch, seq_len) —— 各节点到终点的预测代价
            weights_pred: (batch, seq_len, 6) —— 自适应权重
            current_pred: (batch, seq_len, 3) —— 未来洋流预测
        """
        # 编码
        tokens = self.encode_node(coords, features, time_steps)
        
        # 加入目标信息（每个token都附加目标位置信息）
        goal_embed = self.spatial_embed(goal_coords.float())  # (B, 1, d/2)
        goal_embed = goal_embed.expand(-1, tokens.size(1), -1)
        
        # 目标信息注入（残差方式）
        tokens[:, :, :self.d_model//2] += goal_embed
        
        # Transformer编码
        encoded = self.transformer_encoder(tokens)  # (B, L, d)
        
        # 三个解码头
        h_pred = self.h_head(encoded).squeeze(-1)              # (B, L)
        weights_pred = self.weight_head(encoded)               # (B, L, 6)
        current_pred = self.current_pred_head(encoded)         # (B, L, 3)
        
        return h_pred, weights_pred, current_pred


class PathPlanningLoss(nn.Module):
    """
    多任务联合损失函数
    
    核心设计：
    1. h(n)预测损失 + 可容许性约束
    2. 权重预测损失
    3. 洋流预测损失（辅助任务）
    """
    
    def __init__(self, admissibility_weight=2.0):
        super().__init__()
        self.adm_weight = admissibility_weight
    
    def forward(self, h_pred, h_true, w_pred, w_true, 
                current_pred, current_true):
        """
        h_pred/h_true: 启发函数预测值/真实值
        w_pred/w_true: 权重预测值/真实值
        current_pred/current_true: 洋流预测值/真实值
        """
        # ===== 损失1：h(n)预测精度 =====
        loss_h = nn.MSELoss()(h_pred, h_true)
        
        # ===== 损失2：可容许性惩罚（关键创新）=====
        # h_pred > h_true时强惩罚（高估会破坏A*最优性）
        overestimate = torch.clamp(h_pred - h_true, min=0)
        loss_admissible = (overestimate ** 2).mean()
        
        # ===== 损失3：权重预测 =====
        loss_w = nn.KLDivLoss(reduction='batchmean')(
            torch.log(w_pred + 1e-8), w_true
        )
        
        # ===== 损失4：洋流预测（辅助任务）=====
        loss_current = nn.MSELoss()(current_pred, current_true)
        
        # ===== 总损失 =====
        total_loss = (loss_h + 
                      self.adm_weight * loss_admissible +
                      0.5 * loss_w + 
                      0.3 * loss_current)
        
        return total_loss, {
            'loss_h': loss_h.item(),
            'loss_admissible': loss_admissible.item(),
            'loss_w': loss_w.item(),
            'loss_current': loss_current.item()
        }
```

## 七、完整的数学表达汇总

### 最终代价函数

$f(n) = g(n) + \alpha \cdot h_\theta(n)$

其中各部分定义如下：

### 三级分层累积代价 $g(n)$

$g(n) = g(parent) + \Delta g(n)$

$\Delta g(n) = \begin{cases} \infty & \text{if } \text{HardCheck}(n) = \text{FAIL} \ \underbrace{\lambda_s \left[ w_1 C_{obs}(n,t) + w_2 C_{sig}(n) \right]}_{\text{安全层}} + \underbrace{w_3 C_{eng}(n) + w_4 C_{mot}(n) + w_5 C_{len}(n) + w_6 C_{loc}(n)}_{\text{效率层}} & \text{else} \end{cases}$

$\lambda_s = 2.0$ （安全层优先级放大系数）

### 硬约束集合（任一触发则剪枝）

$\text{HardCheck}(n) = \begin{cases} \text{FAIL} & \text{if } \text{terrain}(n) = 1 \ \text{FAIL} & \text{if } S(n) < S_{crit} \ \text{FAIL} & \text{if } E_{used} + E_{return} > 0.95 \cdot E_{budget} \ \text{FAIL} & \text{if } \min_{o \in O_t} \text{dist}(n,o) < R_{hard} \ \text{FAIL} & \text{if } \Delta\theta(n) > \theta_{max} \ \text{PASS} & \text{otherwise} \end{cases}$

### 各软约束代价项定义

|代价项|公式|值域|
|---|---|---|
|$C_{obs}(n,t)$|$\max_o \exp\left[-\lambda (\text{dist}(n,o) - R_{hard})\right]$ ， $\text{dist} \in [R_{hard}, R_{warn})$ ；否则0|$[0,1]$|
|$C_{sig}(n)$|$\left( \frac{S_{th} - S(n)}{S_{th} - S_{crit}} \right)^2$ ， $S \in [S_{crit}, S_{th})$ ；否则0|$[0,1]$|
|$C_{eng}(n)$|$(1-\beta)\frac{1 - \hat{v} \cdot \hat{d}}{2} + \beta \frac{|\vec{v}|
|$C_{mot}(n)$|$\left( \frac{\Delta\theta}{\pi} \right)^2$|$[0,1]$|
|$C_{len}(n)$|$\frac{\text{step}}{D_{max}}$|$[0,1]$|
|$C_{loc}(n)$|$0.4 \frac{z}{\text{gs}} + 0.6 \frac{\text{dist}_{beacon}}{\text{range}_{beacon}}$|$[0,1]$|

### Transformer学习型启发函数 $h(n)$

$h_\theta(n) = 0.85 \cdot \text{Transformer}_\theta(\text{coords}(n), \text{features}(n), t, \text{goal})$

0.85为可容许性缩放因子

### 自适应权重

$$\vec{W}(t) = (1 - \eta) \vec{W}(t-1) + \eta \cdot \text{Transformer}_W(\text{env}_{\text{state}}(t))$$

$\eta = 0.3$ （指数移动平均平滑系数）

### 核心评估指标

```python
import numpy as np  
from scipy.spatial.distance import euclidean  
​  
class EvaluationMetrics:  
    """路径规划综合评估指标"""  
      
    @staticmethod  
    def compute_all(path, env_data_sequence, config):  
        metrics = {}  
          
        # 1. 路径长度（欧氏总距离）  
        total_dist = 0  
        for i in range(1, len(path)):  
            total_dist += euclidean(path[i], path[i-1])  
        metrics['path_length'] = total_dist  
          
        # 2. 能耗效率（顺流利用率）  
        energy_total = 0  
        for i in range(1, len(path)):  
            t = min(i, len(env_data_sequence)-1)  
            env = env_data_sequence[t]  
            cur_u, cur_v, cur_w = env['current_field']  
            x, y, z = path[i]  
            v = np.array([cur_u[x,y,z], cur_v[x,y,z], cur_w[x,y,z]])  
            d = np.array(path[i]) - np.array(path[i-1])  
            d_norm = np.linalg.norm(d)  
            if d_norm > 0:  
                cos_angle = np.dot(v, d) / (np.linalg.norm(v) * d_norm + 1e-8)  
                energy_total += max(1 - cos_angle, 0.1)  
        metrics['total_energy'] = energy_total  
          
        # 3. 信号合规率  
        compliant = 0  
        for x, y, z in path:  
            if env_data_sequence[0]['signal_field'][x,y,z] >= config['S_th']:  
                compliant += 1  
        metrics['signal_compliance'] = compliant / len(path)  
          
        # 4. 最小障碍距离（安全裕度）  
        min_obs_dist = float('inf')  
        for i, (x,y,z) in enumerate(path):  
            t = min(i, len(env_data_sequence)-1)  
            for obs in env_data_sequence[t]['obstacles_t']:  
                d = euclidean((x,y,z), obs)  
                min_obs_dist = min(min_obs_dist, d)  
        metrics['min_obstacle_dist'] = min_obs_dist  
          
        # 5. 路径平滑度（平均转向角）  
        angles = []  
        for i in range(2, len(path)):  
            v1 = np.array(path[i]) - np.array(path[i-1])  
            v0 = np.array(path[i-1]) - np.array(path[i-2])  
            cos_t = np.dot(v0,v1) / (np.linalg.norm(v0)*np.linalg.norm(v1)+1e-8)  
            angles.append(np.arccos(np.clip(cos_t, -1, 1)))  
        metrics['avg_turn_angle'] = np.mean(angles) if angles else 0  
        metrics['smoothness'] = 1.0 - metrics['avg_turn_angle'] / np.pi  
          
        # 6. 搜索效率（节点扩展数/路径长度）  
        # 需要从stats中获取，此处预留  
          
        return metrics
```

## 十、方案对比总结

|维度|原方案问题|修订方案改进|
|---|---|---|
|代价量纲|∞/100/5/0.2混用，尺度差4个数量级|全部归一化到[0,1]，硬约束独立剪枝|
|因素融合|5大因素平铺加权，目标冲突|三级分层（硬约束→安全→效率），优先级明确|
|动态障碍|安全半径内全设∞，路径僵化|硬半径剪枝+软半径指数衰减，路径可平滑绕行|
|信号惩罚|0/100两档阶梯跳变|连续二次函数，消除代价突变|
|双重惩罚|礁石+障碍+信号三重叠加|同类因素取max，异类因素独立加和|
|权重系统|固定手动设定|场景感知自适应+指数平滑过渡|
|h(n)可容许性|未保证，Transformer可能高估|训练损失加可容许性惩罚+推理缩放0.85|
|洋流建模|仅方向点积|方向代价+强度代价双通道融合|
|地图生成|三重循环慢，膨胀后可能堵死路径|scipy向量化+球形膨胀+起终点清空保证|
|Transformer|仅预测h(n)|双头输出（h(n)+自适应权重），三任务联合训练|
|实验设计|未规划|7级消融实验+5个基线对比+6项指标|

### 核心总结

原方案是"5个因素往一个锅里加"，修订方案是"先过安检门（硬约束），再买保险（安全层），最后比价格（效率层）"——这才是工程上可落地的多约束路径规划设计。

---

### 总结

1. 核心代价函数采用三级分层设计：硬约束（剪枝）→安全层（放大惩罚）→效率层（常规代价），所有软代价均归一化到[0,1]区间，解决量纲混乱问题。
    
2. 关键改进包括：动态障碍的软硬半径结合惩罚、信号约束的连续二次函数、Transformer双头输出（启发函数+自适应权重）及可容许性保障。
    
3. 实验验证体系包含7级消融实验和5类基线对比，核心评估指标覆盖路径长度、能耗、安全性、平滑度等6个维度，参数配置提供明确的推荐值和调参范围。
